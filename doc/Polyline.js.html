class Polyline {
	/**
	 * Constructor
	 * @param {Array} points [[Description]]
	 */
	constructor(points) {
		/** @member {Array} */
		this._points = points || [];
	}
	//*** GETTERS/SETTERS ***************************************
	get points() {
		return this._points;
	}
	set points(pts) {
		this._points = [];
		pts.forEach(function (pt) {
			this._point.push(pt);
		});
	}
	get length() {
		return this._points.length;
	}
	toString() {
		return this._points.join(" ");
	}
	//*** PRIVATE METHODS **************************************

	//*** PUBLIC METHODS ***************************************
	append(pt) {
		this.insertAt(pt, -1);
		return pt;
	}
	prepend(pt) {
		this.insertAt(pt, 0);
		return pt;
	}
	insertAt(pt, idx) {
		idx = (idx === undefined) ? this._points.length : idx;
		this._points.splice(idx, 0, pt);
		pt.polyline = this;
		return pt;
	}
	insertBefore(pt, ref) {
		var idx;
		if (ref) {
			idx = this._points.indexOf(ref);
			if (idx < 0) {
				throw "Reference point doesn't belong to polyline.";
			}
		} else {
			idx = -1;
		}
		this.insertAt(pt, idx);
		return pt;
	}
	indexOf(pt) {
		return this._points(pt);
	}
	//*** STATIC METHODS ***************************************
}
